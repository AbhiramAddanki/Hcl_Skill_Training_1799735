Given an array of integers nums sorted in non-decreasing order, find the starting and ending position of a given target value.

If target is not found in the array, return [-1, -1].

You must write an algorithm with O(log n) runtime complexity.

 
Example 1:

Input: nums = [5,7,7,8,8,10], target = 8
Output: [3,4]

Approach:
  Simialar to binary search but onces we find the target the target may be the first occurence or it there may be one more
  element prior to it,so we should keep the track of mid and search  at left hand side 
  same with last occurence keep the track of right one and search at righthand side.

  code:
  class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int low=first(nums,target);
        int high=last(nums,target);
        vector<int> ans={low,high};
        return ans;
        
    }
    int first(vector<int>& arr,int tar){
        int low=-1;
        int left=0;
        int n=arr.size();
        int right=n-1;
        while(left<=right){
            int  mid=left+(right-left)/2;
            if(arr[mid]==tar){
                low=mid;
                right=mid-1;
           }
           else if(arr[mid]<tar){
            left=mid+1;
           }
           else{
            right=mid-1;
           }
        }
        return low;
    }
    
    int last(vector<int>& arr,int tar){
        int high=-1;
        int left=0;
        int n=arr.size();
        int right=n-1;
        while(left<=right){
            int  mid=left+(right-left)/2;
            if(arr[mid]==tar){
                high=mid;
                left=mid+1;
           }
           else if(arr[mid]<tar){
            left=mid+1;
           }
           else{
            right=mid-1;
           }
        }
        return high;
    }
};
